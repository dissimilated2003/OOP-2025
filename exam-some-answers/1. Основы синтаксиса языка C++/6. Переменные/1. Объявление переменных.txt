//  <спецификаторы> <тип> <индентификатор> 

//нельзя объявлять одинаковые идентификаторы:
int x = 23;
float x = 32;			// ошибка х: переопределение; различные базовые типы
std::cout << x << std::endl;
//------------------------------
//но можно так:
int main()
{
    int x = 23;
    {
        float x = 32;
        std::cout << x << std::endl; 	// выведет 32, это называется "затенение - shadowing"
    }					// берётся ближайшая в текущем scope
}

/*
переменная должна быть проинициализирована при объявлении
без инициализации компилятор будет давать предупреждения
*/
//объявление:		// без выделения памяти
extern int globvar; 	// ключевое слово extern, говорящий о том, что эта переменная/функция 
			// где-то в другом файле. нужен для разделения кода между файлами
			// без extern компилятор пытается создать новую переменную

//определение:
int globvar = 20;	// выделяется память. глобальные переменные инициализируются нулями, указатели = nullptr

/*
локальные, автоматические переменные создаются на стеке
динамически выделяемые переменные создаются в куче
int* pi = new int(42);	// int* - сам указатель на стеке, содержимое *int (42) - в куче
статические переменные - в статической памяти
constexpr может быть выброшено и заменено на литерал (в read-only памяти)
*/
