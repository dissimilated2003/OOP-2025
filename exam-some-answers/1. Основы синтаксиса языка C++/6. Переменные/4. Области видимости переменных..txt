// локальная область видимости -----------------
void DoSomethin()
{
    int x = 10;				// x - локальная переменная, видна внутри функции DoSomethin, вне недоступна
    {
        int y = 10;			// локальная переменная во внутреннем блоке, видна только внутри этого блока
        std::cout << x << std::endl;	// переменная x доступна внутри блока
        for (int k = 5; k--;) x++	// переменная k доступна только внутри цикла for 
    }
    // std::cout << y << std::endl;	// мы вышли из блока, где была объявлена переменная y, она уничтожилась 
					// scope функции ничего о ней не знает
}

// глобальная область видимости ----------------
int glob = 10;				// glob - глобальная переменная, её видят все функции и вся программа

// область класса ------------------------------
class MyClass
{
public:
    void DoSomethin() 
    {
        std::cout << "Pook" << std::endl;
        m_poo++;
    }
private:
    int m_poo{};			// переменная-член класса, доступна только внутри него, его методам
}

// область пространства имён -------------------
namespace Duck
{
    int ducks = 8;
    int sweetDucks = 4;			// видны без проблем только внутри этого пространства имён
					// пространство имён - как большой блок, чтобы получить доступ к его переменным
}					// надо написать его имя и операцию разрешения доступа ::

void DoSomethin2()
{
    Duck::sweetDucks--;		// получение доступа к namespace-scope переменной
}

    int x{5};	// всё, что внутри находится, доступно только внутри текущего файла (внутреннее связывание)
