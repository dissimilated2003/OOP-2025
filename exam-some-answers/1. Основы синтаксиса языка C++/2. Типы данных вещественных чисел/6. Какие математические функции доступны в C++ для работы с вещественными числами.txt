// тригонометрия
constexpr double PI = 3.1415; 
std::cout << "\n";
std::cout << std::sin(PI/2) 	<< "\n";	// sin П/2
std::cout << std::asin(1/3) 	<< "\n";	// arcsin 0.(3)
std::cout << std::cos(0.0) 	<< "\n";	// cos 0
std::cout << std::acos(0.5) 	<< "\n";	// arccos 0.5

// округление

std::cout << "\n";
