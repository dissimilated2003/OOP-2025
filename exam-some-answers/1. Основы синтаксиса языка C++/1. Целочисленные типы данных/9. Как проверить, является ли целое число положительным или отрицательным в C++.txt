void SimpleRoute()
{
    int x = 10;
    int y = -5;
    auto isPositive = [](int n)
    {
        return n >= 0;
    };

    std::cout << (isPositive(x) ? "Pos" : "Neg") << "\n";
    std::cout << (isPositive(y) ? "Pos" : "Neg") << "\n";
}

void BitshiftRoute()
{
    int n = -5;
    std::cout << ((n >> 31) ? "Neg" : "Pos") << "\n";
}

#include <cmath>
void SignbitRoute()
{
    int n = -5;
    std::cout << (std::signbit(n) ? "Neg" : "Pos") << "\n";
}
