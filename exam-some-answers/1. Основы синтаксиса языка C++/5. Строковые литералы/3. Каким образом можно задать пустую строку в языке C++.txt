задать пустую строку можно следующим образом:
---------------------------------------------
const char* str1 = "";
const char str2[] = "";
std::string str3 = ""; // используя STL класс строк

она будет иметь длину равную 0, и содержать только нулевой символ:
------------------------------------------------------------------
#include <cassert>
assert(strlen(str1) == 0);
assert(str1[0] == '\0');
assert(strlen(str2) == 0);
assert(str2[0] == '\0');
assert(str3.length() == 0);
assert(str3[0] == '\0');
std::cout << "successfully\n";
